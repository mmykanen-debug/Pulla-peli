  <script>
    const countDisplay = document.getElementById("count");
    const clickBtn = document.getElementById("clickBtn");
    const upgradeBtn = document.getElementById("upgradeBtn");
    const upgradesDisplay = document.getElementById("upgrades");
    const timeDisplay = document.getElementById("time");
    const cheatBtn = document.getElementById('cheatBtn');
    const cheatInput = document.getElementById('cheatInput');
    const cheatMsg = document.getElementById('cheatMsg');

    let count = 0;
    let mummot = 0;
    let mummo2 = 0, seta = 0, koira = 0, kissa = 0;
    let seconds = 0;
    let prestige = 0;
    let prestigeMultiplier = 1;
    let inventory = { Kurkku:0, Majoneesi:0, Kalja:0, 'Suomi Vodka':0 };
    let history = { Kurkku: [] };
    let costMummo2 = 2;
    let costSeta = 2;
    let costKoira = 2;
    let costKissa = 2;

    const rateMummo2 = 1;
    const rateSeta = 4;
    const rateKoira = 10;
    const rateKissa = 25;

    function updateCounter() {
      countDisplay.textContent = count;
    }

    function updateTime() {
      seconds++;
      const days = Math.floor(seconds / 86400);
      const hours = Math.floor((seconds % 86400) / 3600);
      const minutes = Math.floor((seconds % 3600) / 60);
      const secs = seconds % 60;
      timeDisplay.textContent = `${days}d ${hours}h ${minutes}m ${secs}s`;
    }

    function updateUpgrades() {
      upgradesDisplay.textContent = `Mummoja: ${mummot}`;
    }

    function updateInventoryText() {
      const inv = Object.entries(inventory).filter(([_,v])=>v>0).map(([k,v])=>k+": "+v).join(', ');
      document.getElementById('inventory').textContent = inv || 'Sinulla ei ole tuotteita viel√§.';
    }

    function updatePrices() {
      const rows = [...document.getElementById('marketTable').querySelectorAll('tr')].slice(1);
      let previousPrices = rows.map(r => parseFloat(r.children[1].textContent));
      rows.forEach((row,i)=>{
        const priceCell = row.children[1];
        const changeCell = row.children[2];
        const oldPrice = previousPrices[i];
        const newPrice = Math.max(1, oldPrice * (1 + (Math.random()-0.5)*0.2));
        previousPrices[i] = newPrice;
        priceCell.textContent = newPrice.toFixed(2);
        const change = ((newPrice - oldPrice)/oldPrice*100).toFixed(1);
        changeCell.textContent = change + '%';
      });
      history.Kurkku.push(previousPrices[0]);
      if(history.Kurkku.length>40) history.Kurkku.shift();
      drawChart();
    }

    function drawChart() {
      const ctx = document.getElementById('marketChart').getContext('2d');
      ctx.clearRect(0,0,400,200);
      const data = history.Kurkku;
      if(data.length<2) return;
      const max = Math.max(...data);
      const min = Math.min(...data);
      data.forEach((value, index) => {
        const x = (index / data.length) * 400;
        const y = 200 - ((value - min) / (max - min)) * 200;
        ctx.fillStyle = 'blue';
        ctx.fillRect(x, y, 2, 2);
      });
    }

    function totalProduction() {
      return (
        mummot * 1 +
        mummo2 * rateMummo2 +
        seta * rateSeta +
        koira * rateKoira +
        kissa * rateKissa
      ) * prestigeMultiplier;
    }

    function prestigeReset() {
      prestige++;
      prestigeMultiplier = 1 + prestige * 0.5;
      count = 0;
      mummot = mummo2 = seta = koira = kissa = 0;
      costMummo2 = costSeta = costKoira = costKissa = 2;
      updateRecruitUI();
      updateCounter();
    }

    function tryBuy(cost, addCallback) {
      if(count >= cost) {
        count -= cost;
        updateCounter();
        addCallback();
      }
    }

    function buyUnit(unit, costVarName, countSpanId) {
      const amt = parseInt(document.getElementById('buyAmount').value, 10);
      for(let i=0;i<amt;i++){
        if(count >= window[costVarName]){
          count -= window[costVarName];
          window[unit] += 2;
          window[costVarName]++;
        }
      }
      updateCounter();
      document.getElementById(countSpanId).textContent = window[unit];
      updateRecruitUI();
    }

    function updateRecruitUI() {
      document.getElementById('buyMummo2').textContent = `Osta Mummo (${costMummo2}‚Ç¨)`;
      document.getElementById('buySeta').textContent = `Osta Set√§ (${costSeta}‚Ç¨)`;
      document.getElementById('buyKoira').textContent = `Osta Koira (${costKoira}‚Ç¨)`;
      document.getElementById('buyKissa').textContent = `Osta Kissa (${costKissa}‚Ç¨)`;
    }

    function applyCheat(code) {
      const normalized = (code || '').trim().toLowerCase();
      if (normalized.match(/^mummo(\d+)$/)) {
        const amount = parseInt(normalized.match(/^mummo(\d+)$/)[1], 10);
        count += amount * 1000000000;
        updateCounter();
        cheatMsg.textContent = 'Lis√§tty ' + amount + ' miljardia pullaa!';
      } else {
        cheatMsg.textContent = 'Virheellinen koodi. K√§yt√§ muotoa mummo1, mummo2...';
      }
    }

    clickBtn.addEventListener("click", () => {
      count++;
      updateCounter();
    });

    upgradeBtn.addEventListener("click", () => {
      if (count >= 10) {
        count -= 10;
        mummot++;
        updateCounter();
        updateUpgrades();
      }
    });

    document.getElementById("buyMummo2").addEventListener('click', ()=>buyUnit('mummo2','costMummo2','mummo2Count'));
    document.getElementById("buySeta").addEventListener('click', ()=>buyUnit('seta','costSeta','setaCount'));
    document.getElementById("buyKoira").addEventListener('click', ()=>buyUnit('koira','costKoira','koiraCount'));
    document.getElementById("buyKissa").addEventListener('click', ()=>buyUnit('kissa','costKissa','kissaCount'));

    cheatBtn.addEventListener('click', () => applyCheat(cheatInput.value));
    cheatInput.addEventListener('keydown', (e) => {
      if (e.key === 'Enter') applyCheat(cheatInput.value);
    });

    setInterval(updateTime, 1000);
    setInterval(() => {
      count += totalProduction();
      updateCounter();
    }, 1000);
    setInterval(updatePrices, 60000);
    setInterval(()=>{
      count += mummot + seta + koira + kissa;
      updateCounter();
    },1000);
  </script>
